var main = function(obj) {
  	
		window.obj = obj; 		
 		
  		obj.set("l1", swell.List.create());
  		  		
		obj.set("n", 1001);	
		obj.set("s", "hello world");
  	
  		obj.set("m1", swell.Map.create());
  		
  		obj.set("m1.n", 1002);
  		
  		obj.node("m1").set("s", "this is a string");
  		obj.node("m1").get();
  		
  		obj.node("m1").set("s", "a different  string");
  		
  		obj.set("m1", 9991);
  		
  		// a map in a list

  		// lists can't set properties
  		// obj.node("l1").set("mm", swell.Map.create());  		 	

		obj.node("l1").add(swell.Map.create());

		// accessing list items
		obj.node("l1").pick(0);
		obj.node("l1.0");	
		
		obj.node("l1.0").set("p1", "value 1");	
		obj.node("l1.0").set("p2", 2);
		obj.node("l1.0").set("p3", true);		
		obj.node("l1").get("0");
		
		obj.node("l1").add("trash");
		obj.node("l1").add("element");
				
		obj.node("l1").add("replacing list at #0", 0);
		
  	};
  	
		var service = swell.runtime.get();
		service.open({
			id: "o1"
		}, {
			onError: function(e) { console.log("Exception: "+e); },
			onSuccess: main			  			
		});
